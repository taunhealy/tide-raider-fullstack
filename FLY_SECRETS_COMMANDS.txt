# Fly.io Backend Secrets - Copy and paste these commands

# Replace the placeholder values with your actual values

# Google OAuth
fly secrets set GOOGLE_CLIENT_ID="YOUR_GOOGLE_CLIENT_ID" --app tide-raider-backend

fly secrets set GOOGLE_CLIENT_SECRET="YOUR_GOOGLE_CLIENT_SECRET" --app tide-raider-backend

# Backend URL (your Fly.io backend URL)
fly secrets set BACKEND_URL="https://tide-raider-backend.fly.dev" --app tide-raider-backend

# Frontend URL (your Vercel frontend URL)
fly secrets set FRONTEND_URL="https://www.tideraider.com" --app tide-raider-backend

# JWT Secret (must match frontend - use your existing NEXTAUTH_SECRET value)
# Generate a new one: node -e "console.log(require('crypto').randomBytes(32).toString('hex'))"
fly secrets set NEXTAUTH_SECRET="YOUR_SECRET_HERE" --app tide-raider-backend

# Node Environment
fly secrets set NODE_ENV="production" --app tide-raider-backend
